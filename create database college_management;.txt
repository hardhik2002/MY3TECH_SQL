create database college_management;
use college_management;
CREATE TABLE college_details (
    college_id INT PRIMARY KEY,           
    college_name VARCHAR(255) NOT NULL,
    address VARCHAR(255),
    city VARCHAR(100),
    state VARCHAR(100),
    postal_code VARCHAR(20),
    established_year year,
    contact_number VARCHAR(15),
    website VARCHAR(100)
);
CREATE TABLE course_details (
    course_id INT PRIMARY KEY,             -- Primary key for course identification
    course_name VARCHAR(255) NOT NULL,
    course_duration INT NOT NULL,          -- Duration in years
    course_fee DECIMAL(10, 2),
    college_id INT,                        -- Foreign key from college_details
    CONSTRAINT fk_college FOREIGN KEY (college_id) REFERENCES college_details(college_id)
);
CREATE TABLE branch_details (
    branch_id INT PRIMARY KEY,             -- Primary key for branch identification
    branch_name VARCHAR(255) NOT NULL,
    course_id INT,                         -- Foreign key from course_details
    branch_head VARCHAR(255),              -- Employee who is head of the branch
    employee_id INT,                       -- Foreign key from employee_details
    CONSTRAINT fk_course FOREIGN KEY (course_id) REFERENCES course_details(course_id),
    CONSTRAINT fk_employee FOREIGN KEY (employee_id) REFERENCES employee_details(employee_id)
);
CREATE TABLE student_details (
    student_id INT PRIMARY KEY,            -- Primary key for student identification
    student_name VARCHAR(255) NOT NULL,
    gender CHAR(1),                        -- M or F
    date_of_birth DATE,
    phone_number VARCHAR(15),
    email VARCHAR(100),
    branch_id INT,                         -- Foreign key from branch_details
    course_id INT,                         -- Foreign key from course_details
    college_id INT,                        -- Foreign key from college_details
    enrollment_year YEAR,
    graduation_year YEAR,
    CONSTRAINT fk_branch FOREIGN KEY (branch_id) REFERENCES branch_details(branch_id),
    CONSTRAINT fk_course FOREIGN KEY (course_id) REFERENCES course_details(course_id),
    CONSTRAINT fk_college FOREIGN KEY (college_id) REFERENCES college_details(college_id)
);
CREATE TABLE employee_details (
    employee_id INT PRIMARY KEY,            -- Primary key for employee identification
    employee_name VARCHAR(255) NOT NULL,
    gender CHAR(1),                         -- M or F
    date_of_birth DATE,
    phone_number VARCHAR(15),
    email VARCHAR(100),
    hire_date DATE,
    position VARCHAR(255),                  -- Job position (e.g., Professor, Admin)
    department VARCHAR(100),
    salary DECIMAL(10, 2),
    college_id INT,                         -- Foreign key from college_details
    CONSTRAINT fk_college FOREIGN KEY (college_id) REFERENCES college_details(college_id)
);
